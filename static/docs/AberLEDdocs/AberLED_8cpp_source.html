112 static void fastShiftOutRows(byte n)
113 {
114  PORTD &= ~(1 << 2); // data off
115  for (int i = 7; i >= 0; i--)
116  {
117  PORTD &= ~(1 << 4); // clock off
118  if (n & (1 << i))
119  PORTD |= 1 << 2; // data on if true
120  else
121  PORTD &= ~(1 << 2); // data off if true
122  PORTD |= 1 << 4; // clock on
123 
124  PORTD &= ~(1 << 2); // data off (to prevent bleed through)
125  }
126  PORTD &= ~(1 << 4); // clock off
127 }

129 static void fastShiftOutCols(uint16_t n)
130 {
131  PORTD &= ~(1 << 5); // data off

139  PORTD |= 1 << 6; // clock on
140  PORTD &= ~(1 << 5); // data off (to prevent bleed through)
141  }

319 // buffer data out to the screen. The front buffer is the one

322 static int refrow = 0;
323 inline void refreshNextRow()
324 {
325  if(isTFT) {
326  // using a TFT, so we draw rectangles instead of using the shift registers (which
327  // don't exist on the TFT boards)
328 
329  uint16_t v = *(frontBuffer + refrow); // get row pointer
330 // uint16_t vold = *(backBuffer + refrow); // get row pointer
331 
332  for(int x=0;x<8;x++){
333  uint16_t q = (v >> (x*2)) & 3;
334  tft.fillRect(16*x+2, 16*refrow+2, 12, 12, cols[q]);
335  }
336  // when we start a new screen, we should add any text to the bottom
337  if(!refrow){
338  if(txtChanged){
339  tft.fillRect(0, 150, 128, 16, TFT_BLACK);
340  txtChanged = false;
341  }
342  if(txtToWrite[0]){
343  tft.setTextColor(TFT_WHITE);
344  tft.setCursor(4, 150);
345  tft.print(txtToWrite);
346  }
347  }
348  } else {
349  // this code is used for the older LED boards, and use the shift registers -
350  // we directly manipulate the port registers for speed.
351 
352  if (!refrow)
353  PORTD |= 1 << 2; // turn on the row data line to get the first bit set
354 
355  // set latches low
356  PORTD &= ~((1 << 3) | (1 << 7));
357 
358  // tick the row clock to move the next bit in (high on
359  // the first row, low after that)
360  PORTD |= (1 << 4);
361  PORTD &= ~(1 << 4);
362 
363  // and turn off the row data line
364  PORTD &= ~(1 << 2);
365 
366  // now the appropriate row is set high, set the column
367  // bits low for the pixels we want.
368 
369  fastShiftOutCols(~(frontBuffer[refrow]));
370  // and latch the registers
371 
372  PORTD |= ((1 << 3) | (1 << 7));
373  }
374 
375  refrow = (refrow + 1) % 8;
376 }

431  break;
432  case 3:

434  break;

437  break;
438  }
439  }
440  else

446  break;
447  case 1:
448  bstate = PINC & 8;
449  break;

452  break;

455  break;

463

478 }

483 static void setupInterrupt()
484 {
485  cli(); // disable all interrupts
486 
487  // some very hardware-specific code, setting registers
488  // inside the ATMega328p processor.
489 
490  // set timer1 interrupt at 500Hz or slower for the TFT
491  TCCR1A = 0; // set entire TCCR1A register to 0
492  TCCR1B = 0; // same for TCCR1B
493  TCNT1 = 0; // initialize counter value to 0
494  if(isTFT) {
495  // 200 Hz
496  OCR1A = 1249;
497  TCCR1B |= (1 << WGM12);
498  TCCR1B |= (1 << CS11) | (1 << CS10);
499  } else {
500  // set compare match register for correct frequency
501  // (16*10^6) / (500*8) - 1 gives 3999 (for 500Hz)
502  OCR1A = 3999;
503  // turn on CTC mode
504  TCCR1B |= (1 << WGM12);
505  // Set prescaler to 8
506  TCCR1B |= (1 << CS11);
507  }
508  // enable timer compare interrupt
509  TIMSK1 |= (1 << OCIE1A);
510  interruptRunning = true;
511  sei(); // re-enable interrupts
512 }
